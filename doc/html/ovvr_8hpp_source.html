79
80 // Init VR system:
81 std::cout << "Using OpenVR " << vr::k_nSteamVRVersionMajor << "." << vr::k_nSteamVRVersionMinor << "." << vr::k_nSteamVRVersionBuild << std::endl;
82 vrSys = vr::VR_Init(&error, vr::VRApplication_Scene);
83 if (error != vr::VRInitError_None)
84 {
85 vrSys = nullptr;
86 std::cout << "[ERROR] Unable to init VR runtime: " << vr::VR_GetVRInitErrorAsEnglishDescription(error) << std::endl;
87 return false;
88 }
89
90 // Init render models:
91 vrModels = (vr::IVRRenderModels*)vr::VR_GetGenericInterface(vr::IVRRenderModels_Version, &error);
92 if (vrModels == nullptr)
93 {
94 vrSys = nullptr;
95 vr::VR_Shutdown();
96 std::cout << "[ERROR] Unable to get render model interface: " << vr::VR_GetVRInitErrorAsEnglishDescription(error) << std::endl;
97 return false;
98 }
99
100 // Initialize the compositor:
101 vrComp = vr::VRCompositor();
102 if (!vrComp)
103 {
104 vrModels = nullptr;
105 vrSys = nullptr;
106 vr::VR_Shutdown();
107 std::cout << "[ERROR] Unable to get VR compositor" << std::endl;
108 return false;
109 }
110
111 // Tweaks:
112 //vrComp->ShowMirrorWindow();
113
114 // Init controller tracking:
115 controllers.clear();
116 for (unsigned int c = 0; c < vr::k_unMaxTrackedDeviceCount; c++)
117 {
118 if (vrSys->GetTrackedDeviceClass(c) == vr::TrackedDeviceClass_Controller)
119 {
120 std::cout << " Found controller at " << c << std::endl;
121 Controller* cont = new Controller();
122 cont->id = c;
123
124 unsigned int bufferLen = vrSys->GetStringTrackedDeviceProperty(c, vr::Prop_RenderModelName_String, nullptr, 0, nullptr);
125 if (bufferLen == 0)
126 {
127 std::cout << "[ERROR] Unable to get controller render model" << std::endl;
128 delete cont;
129 continue;
130 }
131
132 std::string result;
133 result.resize(bufferLen);
134 vrSys->GetStringTrackedDeviceProperty(c, vr::Prop_RenderModelName_String, &result[0], bufferLen, nullptr);
135 std::cout << " Controller render model: '" << result << "'" << std::endl;
136 controllers.push_back(cont);
137 }
138 }
139
140 // Done:
141 return true;
142 }
